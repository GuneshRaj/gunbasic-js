GunBasic UI Component Showcase

Complete SwiftUI-inspired component library

VStack - Vertical Layout
var layout = VStack([
    Text("Item 1"),
    Text("Item 2"), 
    Text("Item 3")
]);
HStack - Horizontal Layout
var layout = HStack([
    Text("Left"),
    Spacer(),
    Text("Right")
]).justify("between");
Text Components
var textDemo = VStack([
    Text("Title Text").font("title"),
    Text("Headline Text").font("headline"),
    Text("Muted Text").color("muted"),
    Badge("Success").style("success"),
    Badge("Pill Badge").style("primary").pill(true)
]);
Text Fields
var form = VStack([
    TextField("Name", "name").placeholder("Enter name"),
    TextField("Email", "email").type("email"),
    TextField("Password", "pass").type("password"),
    TextArea("Message", "msg").rows(3)
]);
Selection Controls
var controls = VStack([
    RadioGroup("Choose", "option", [
        {value: "1", label: "Option 1"},
        {value: "2", label: "Option 2"}
    ]),
    Select("Select Item", "item", ["A", "B", "C"]),
    Checkbox("I agree", "agree"),
    Toggle("Enable", "notify")
]);
Buttons
var buttons = VStack([
    Button("Primary", "test").style("primary"),
    Button("Secondary", "test").style("secondary"),
    Button("Large", "test").style("success").size("lg")
]);
Dropdown Menu
var dropdown = DropdownMenu("Actions", [

    {label: "Delete", action: "test"},

Card Component
var card = Card().children([
    Text("Card Title").font("title"),
    Text("Card content here"),
    Button("Action", "test").style("primary")
]).header("Header").footer("Footer");
Alert Component
var alerts = VStack([
    Alert("Success message").style("success"),
    Alert("Warning message").style("warning"),
    Alert("Error message").style("danger")
]);
Complete Application Example
var app = VStack([
    Card().children([
        HStack([
            Text("Dashboard").font("title"),
            Spacer(),
            Badge("Online").style("success")
        ])
    ]).header("App Header"),
    
    HStack([
        Card().children([
            Text("Navigation").font("headline"),
            VStack([
                Button("Home", "test").style("outline"),
                Button("Users", "test").style("outline")
            ])
        ]),
        Card().children([
            Text("User Form").font("title"),
            TextField("Name", "name").placeholder("Enter name"),
            TextField("Email", "email").type("email"),
            Button("Submit", "test").style("primary")
        ])
    ])
]);
